def creat_time_series(num_features, num_act_labels, num_id_labels, label_codes, trial_codes):
    dataset_columns = num_features+num_act_labels+num_id_labels
    ds_list = get_ds_infos()
    train_data = np.zeros((0,dataset_columns))
    test_data = np.zeros((0,dataset_columns))
    feature = "userAcceleration"
    axes = [feature+".x", feature+".y", feature+".z"]
    for i, sub_id in enumerate(ds_list[:,0]):
        for j, act in enumerate(label_codes):
            for trial in trial_codes[act]:
                fname = 'A_DeviceMotion_data/'+act+'_'+str(trial)+'/sub_'+str(int(sub_id))+'.csv'
                raw_data = pd.read_csv(fname)
                raw_data = raw_data.drop(['Unnamed: 0'], axis=1)
                raw_data = (raw_data[axes]**2).sum(axis=1)**0.5
                unlabel_data = raw_data[:,np.newaxis] 
                label_data = np.zeros((len(unlabel_data), dataset_columns))
                label_data[:,:-(num_act_labels + num_id_labels)] = unlabel_data
                label_data[:,label_codes[act]] = 1
                label_data[:,int(sub_id) - (num_id_labels+1)] = 1
                ## We consider long trials as training dataset and short trials as test dataset
                if trial > 10:
                    test_data = np.append(test_data, label_data, axis = 0)
                else:    
                    train_data = np.append(train_data, label_data, axis = 0)
    return train_data , test_data

def time_series_to_section(dataset, num_act_labels, num_id_labels, sliding_window_size, step_size_of_sliding_window, standardize = False, **options):
    data = dataset[: , 0:-(num_act_labels+num_id_labels)]
    act_labels = dataset[: , -(num_act_labels+num_id_labels):-(num_id_labels)]
    id_labels = dataset[: , -(num_id_labels):]
    mean = 0
    std = 1
    
    if standardize:
        ## Standardize each sensors data to have a zero mean and unity standard deviation.
        ## As usual, we normalize test dataset by training dataset's parameters 
        if options:
            mean = options.get("mean")
            std = options.get("std")
            print("----> Test Data has been standardized")
        else:
            mean = data.mean(axis=0)
            std = data.std(axis=0)
            print("----> Training Data has been standardized:\n the mean is = ",str(mean.mean())," ; and the std is = ",str(std.mean()))            
  
        data -= mean
        data /= std
    else:
        print("----> Without Standardization.....")

    ## We want the Rows of matrices show each Feature and the Columns show time points.
    data = data.T
            
    size_features = data.shape[0]
    size_data = data.shape[1]
    number_of_secs = round(((size_data - sliding_window_size)/step_size_of_sliding_window))
            
    ##  Create a 3D matrix for Storing Snapshots  
    secs_data = np.zeros((number_of_secs , size_features , sliding_window_size ))
    act_secs_labels = np.zeros((number_of_secs, num_act_labels))
    id_secs_labels = np.zeros((number_of_secs, num_id_labels))
    
    k=0    
    for i in range(0 ,(size_data)-sliding_window_size  , step_size_of_sliding_window):
        j = i // step_size_of_sliding_window
        if(j>=number_of_secs):
            break
        if(not (id_labels[i] == id_labels[i+sliding_window_size-1]).all()): 
            continue
        if(not (act_labels[i] == act_labels[i+sliding_window_size-1]).all()): 
            continue    
        secs_data[k] = data[0:size_features, i:i+sliding_window_size]
        act_secs_labels[k] = act_labels[i].astype(int)
        id_secs_labels[k] = id_labels[i].astype(int)
        k = k+1
    secs_data = secs_data[0:k]
    act_secs_labels = act_secs_labels[0:k]
    id_secs_labels = id_secs_labels[0:k]
    print(k)
    return secs_data, act_secs_labels, id_secs_labels, mean, std

class Estimator:
    @staticmethod
    def shared(inp):
        # Start
        x = Conv2D(64, (1, 5), padding="same", activation='relu')(inp)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        x = Dropout(0.25)(x)

        # 1
        x = Conv2D(64, (1, 5), padding="same", activation='relu')(x)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        x = Dropout(0.25)(x)
        
        # 2
        x = Conv2D(64, (1, 3), padding="same", activation='relu')(x)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        out = Dropout(0.25)(x)
        
        return out
    
    @staticmethod
    def sensitive(inp, num_classes, act_func="softmax", b_name="Sensitive"):
        # 3
        x = Conv2D(128, (1, 3), padding="same", activation='relu')(inp)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        x = Dropout(0.25)(x)
        
        # 4
        x = Conv2D(128, (1, 3), padding="same", activation='relu')(x)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        x = Dropout(0.25)(x)
        
        # End
        x = Flatten()(x)
        x = Dense(512, activation='relu')(x)
        x = Dropout(0.5)(x)
        out = Dense(num_classes, activation=act_func, name = b_name)(x)

        return out
    @staticmethod
    def nonsensitive(inp, num_classes, act_func="softmax", b_name="NonSensitive"):
        # 3
        x = Conv2D(128, (1, 3), padding="same", activation='relu')(inp)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        x = Dropout(0.25)(x)
        
        # 4
        x = Conv2D(128, (1, 3), padding="same", activation='relu')(x)
        x = MaxPooling2D(pool_size=(1, 2))(x)
        x = Dropout(0.25)(x)
        
        # End
        x = Flatten()(x)
        x = Dense(512, activation='relu')(x)
        x = Dropout(0.5)(x)
        out = Dense(num_classes, activation=act_func, name = b_name)(x)

        return out
    @staticmethod
    def build(height, width, num_sen_classes, num_nonsen_classes, sen_act_func="softmax", nonsen_act_func="softmax"):
        # Estimator
        inp = Input(shape=(height, width, 1))
        shared = Estimator.shared(inp)
        nonsen = Estimator.nonsensitive(shared, num_nonsen_classes, act_func=nonsen_act_func)
        sen = Estimator.sensitive(shared, num_sen_classes, act_func=sen_act_func)
        model = Model(inputs=inp, outputs=[nonsen,sen],name="Estimator")
        return model
